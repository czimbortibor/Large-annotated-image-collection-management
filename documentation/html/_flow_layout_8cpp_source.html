107 qreal FlowLayout::doLayout(const QRectF &geom, bool applyNewGeometry) const {
108  qreal left, top, right, bottom;
109  getContentsMargins(&left, &top, &right, &bottom);
110  const qreal maxw = geom.width() - left - right;
111 
112  qreal x = 0;
113  qreal y = 0;
114  qreal maxRowHeight = 0;
115  QSizeF pref;
116  for (int i = 0; i < _items.count(); ++i) {
117  QGraphicsLayoutItem *item = _items.at(i);
118  pref = item->effectiveSizeHint(Qt::PreferredSize);
119  maxRowHeight = qMax(maxRowHeight, pref.height());
120 
121  qreal next_x;
122  next_x = x + pref.width();
123  if (next_x > maxw) {
124  if (x == 0) {
125  pref.setWidth(maxw);
126  } else {
127  x = 0;
128  next_x = pref.width();
129  }
130  y += maxRowHeight + spacing(Qt::Vertical);
131  maxRowHeight = 0;
132  }
133 
134  if (applyNewGeometry)
135  item->setGeometry(QRectF(QPointF(left + x, top + y), pref));
136  x = next_x + spacing(Qt::Horizontal);
137  }
138  maxRowHeight = qMax(maxRowHeight, pref.height());
139  return top + y + maxRowHeight + bottom;
140 }

142 QSizeF FlowLayout::minSize(const QSizeF &constraint) const {
143  QSizeF size(0, 0);
144  qreal left, top, right, bottom;
145  getContentsMargins(&left, &top, &right, &bottom);
146  if (constraint.width() >= 0) { // height for width
147  const qreal height = doLayout(QRectF(QPointF(0,0), constraint), false);
148  size = QSizeF(constraint.width(), height);
149  } else if (constraint.height() >= 0) { // width for height?
150  // not supported
151  } else {
152  QGraphicsLayoutItem *item;
153  foreach (item, _items)
154  size = size.expandedTo(item->effectiveSizeHint(Qt::MinimumSize));
155  size += QSize(left + right, top + bottom);
156  }
157  return size;
158 }

160 QSizeF FlowLayout::prefSize() const {
161  qreal left, right;
162  getContentsMargins(&left, 0, &right, 0);
163 
164  QGraphicsLayoutItem* item;
165  qreal maxh = 0;
166  qreal totalWidth = 0;
167  foreach (item, _items) {
168  if (totalWidth > 0)
169  totalWidth += spacing(Qt::Horizontal);
170  QSizeF pref = item->effectiveSizeHint(Qt::PreferredSize);
171  totalWidth += pref.width();
172  maxh = qMax(maxh, pref.height());
173  }
174  maxh += spacing(Qt::Vertical);
175 
176  const qreal goldenAspectRatio = 1.61803399;
177  qreal w = qSqrt(totalWidth * maxh * goldenAspectRatio) + left + right;
178 
179  return minSize(QSizeF(w, -1));
180 }

182 QSizeF FlowLayout::maxSize() const {
183  QGraphicsLayoutItem* item;
184  qreal totalWidth = 0;
185  qreal totalHeight = 0;
186  foreach (item, _items) {
187  if (totalWidth > 0)
188  totalWidth += spacing(Qt::Horizontal);
189  if (totalHeight > 0)
190  totalHeight += spacing(Qt::Vertical);
191  QSizeF pref = item->effectiveSizeHint(Qt::PreferredSize);
192  totalWidth += pref.width();
193  totalHeight += pref.height();
194  }
195 
196  qreal left, top, right, bottom;
197  getContentsMargins(&left, &top, &right, &bottom);
198  return QSizeF(left + totalWidth + right, top + totalHeight + bottom);
199 }
